import { BodyText, Box, TinyText } from 'cabana-react';
import * as icons from 'cabanaico';
export const Icons = () => (
    <Box
        fontSize="lead"
        display="grid"
        color="darkGrey"
        style={{ gridTemplateRows: 'repeat(30, 1fr)', gridTemplateColumns: 'repeat(5, 1fr)', gap: '30px' }}
    >
        {Object.keys(icons).map(key => {
            const Icon = icons[key];
            return (
                <Box title={key} display="inline" >
                    <Icon key={key} />
                    <TinyText color="grey" display="block">{key}</TinyText>
                </Box>
            );
        })}
    </Box>
);
